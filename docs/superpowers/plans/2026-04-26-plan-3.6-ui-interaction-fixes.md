**Prerequisite:** Plan 3.5 complete on `main` (commit `4b102aa`). UI is the single-page event-focused dashboard. 85 passed, 4 skipped baseline.

**Goal:** Fix three concrete UX failures shipped in Plan 3.5: (1) price-chart marker clicks don't fire, (2) event-axis labels stack on top of each other, (3) ripple-tree node clicks don't propagate news into the timeline. After this plan, clicking a marker explains the day, the timeline reads as a staggered chronology, and clicking a ripple sector replaces the timeline with that sector's narrative.

- Task 1 swaps Streamlit's native `st.plotly_chart(on_select=...)` (which only fires on box/lasso selection, not single clicks) for `streamlit-plotly-events.plotly_events(click_event=True)` (real click handler). Hooks the existing `selected_date` session-state key — no downstream changes.

- Task 3 adds a `selected_sector` session-state key set by `agraph()`'s return value in `ripple.render`, and a "sector-driven" branch in `event_axis.render` that swaps the price-derived markers for the clicked node's `supporting_news`. Adds a "← Back to price view" button in the event-axis zone. Modifies `tree_to_graph_elements` to also return an `id → node_data` map so we can resolve the agraph click back to the original tree node.

### Task 1: Real click handler via `streamlit-plotly-events`

**Why:** Plan 3.5 used `st.plotly_chart(fig, on_select="rerun", selection_mode="points")`. _MARKERS_CURVE_INDEX = 1  # 0=line trace, 1=markers trace (build_figure adds them in this order)


def _click_event_to_iso(events: list, moves: List[Dict]) -> Optional[str]:
    """Map a plotly_events click result to an ISO date string.

    Returns None if no event, the click is on the line (not markers), or
    the pointIndex is out of range. Defensive against the upstream
    streamlit-plotly-events shape changing in a minor version."""
    if not events:
        return None
    e = events[0]
    if e.get("curveNumber") != _MARKERS_CURVE_INDEX:
        return None
    idx = e.get("pointIndex")
    if not isinstance(idx, int) or idx < 0 or idx >= len(moves):
        return None
    return moves[idx]["date"]
```

_LABEL_Y_ABOVE = 1.55
_LABEL_Y_BELOW = 0.45


def _label_y_for_index(i: int) -> tuple:
    """Return (y_position, yanchor) for the label of the i-th marker.
    Even index → label above the baseline (y=1.55, anchored at the bottom of
    the text box). Odd index → label below (y=0.45, anchored at the top)."""
    if i % 2 == 0:
        return (_LABEL_Y_ABOVE, "bottom")
    return (_LABEL_Y_BELOW, "top")


def _build_figure(dates_with_headlines: List[Dict], window_start: date, window_end: date) -> go.Figure:
    """Staggered horizontal timeline: baseline at y=1, vertical stems connect
    each marker to a label box positioned alternately above and below.
    Form factor inspired by classic chronology diagrams."""
    fig = go.Figure()

    # 1) Horizontal baseline axis.
    fig.add_shape(
        type="line",
        x0=pd.Timestamp(window_start), x1=pd.Timestamp(window_end),
        y0=1, y1=1,
        line=dict(color="#bbbbbb", width=2),
    )

    # 2) Per-marker: stem + label annotation. Markers themselves go in one
    #    Scatter trace so hover works uniformly.
    xs: List = []
    ys: List = []
    colors: List = []
    hovers: List = []
    for i, d in enumerate(dates_with_headlines):
        x = pd.Timestamp(d["date"])
        color = _UP_COLOR if d["direction"] == "up" else _DOWN_COLOR
        label_y, yanchor = _label_y_for_index(i)

        # Vertical stem from baseline (y=1) up/down to label_y.
        fig.add_shape(
            type="line",
            x0=x, x1=x, y0=1, y1=label_y,
            line=dict(color=color, width=1.2),
            opacity=0.6,
        )
        # Label annotation at the end of the stem, with a subtle border.
        fig.add_annotation(
            x=x, y=label_y,
            text=_truncate(d["label"], 28),
            showarrow=False,
            xanchor="center", yanchor=yanchor,
            font=dict(size=10, color="#222"),
            bgcolor="rgba(255,255,255,0.9)",
            bordercolor=color,
            borderwidth=1,
            borderpad=4,
        )
        xs.append(x)
        ys.append(1)
        colors.append(color)
        hovers.append(d["hover"])

    # 3) Markers on the baseline (single trace for hover uniformity).
    fig.add_trace(go.Scatter(
        x=xs, y=ys, mode="markers",
        marker=dict(color=colors, size=14, line=dict(color="white", width=1.5)),
        hovertext=hovers, hoverinfo="text",
        showlegend=False,
    ))

    fig.update_layout(
        title=dict(text="Event narrative axis", font=dict(size=14)),
        height=320,
        margin=dict(l=40, r=40, t=50, b=20),
        xaxis=dict(showgrid=False),
        yaxis=dict(visible=False, range=[0, 2]),
        showlegend=False,
    )
    return fig
```

def tree_to_graph_elements(tree: Dict) -> Tuple[List[Node], List[Edge], Dict[str, Dict]]:
    """Walk the tree and emit (nodes, edges, id_map).

    id_map maps each auto-generated node id (e.g. "n1", "n2", ...) to the
    original tree node dict — used by render() to resolve agraph()'s clicked
    node id back to the underlying sector data (supporting_news, mechanism,
    severity). The "root" id is intentionally NOT in id_map (the root is
    just the event title; clicks on it should not switch the timeline)."""
    nodes: List[Node] = [Node(id="root", label=tree.get("event", "Event"), size=25, color="#1976d2")]
    edges: List[Edge] = []
    id_map: Dict[str, Dict] = {}
    counter = {"i": 0}

    def _walk(children: List[Dict], parent_id: str) -> None:
        for n in children:
            counter["i"] += 1
            nid = f"n{counter['i']}"
            sev = n.get("severity", "moderate")
            color = _SEVERITY_COLORS.get(sev, "#9e9e9e")
            size = _SEVERITY_SIZES.get(sev, 14)
            nodes.append(Node(id=nid, label=_label(n), size=size, color=color,
                              title=_tooltip(n)))
            edges.append(Edge(source=parent_id, target=nid))
            id_map[nid] = n
            _walk(n.get("children", []), nid)

    _walk(tree.get("nodes", []), "root")
